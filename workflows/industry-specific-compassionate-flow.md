// Function to get industry-specific compassionate variation
function getIndustryCompassionateVariation(industry, category) {
  const industryVariations = industryCompassionateVariations[industry];
  if (!industryVariations || !industryVariations[category]) {
    // Fallback to generic compassionate variations
    return getCompassionateVariation(category);
  }
  const variations = industryVariations[category];
  return variations[Math.floor(Math.random() * variations.length)];
}

// Function to format industry-specific response with dynamic data
function formatIndustryCompassionateResponse(template, data = {}) {
  let response = template;
  Object.keys(data).forEach(key => {
    response = response.replace(new RegExp(`{${key}}`, 'g'), data[key]);
  });
  return response;
}
